def update_bios(replace_existing=True) -> Union[pd.DataFrame,None]:
    """Update player bios in the library's CSV files
    
    Parameters:
    -----------
    replace_existing : bool default TRUE
        if FALSE, function will simply return the data retrieved from the API without updating the current CSV file
        
    """
    
    df = get_bios_async(include_list_data=True)
    df.rename(columns={
        'name':'name_full',
        'nameFirst':'name_first',
        'nameMiddle':'name_middle',
        'nameLast':'name_last',
        'nameGiven':'name_given',
        'nameBoxscore':'name_boxscore',
        'nameMatrilineal':'name_matrilineal',
        'nickName':'name_nick',
        'birthDate':'birth_date',
        'birthCity':'birth_city',
        'birthStateProvince':'birth_state',
        'birthCountry':'birth_country',
        'deathDate':'death_date',
        'deathCity':'death_city',
        'deathStateProvince':'death_state',
        'deathCountry':'death_country',
        'isPlayer':'is_player',
        'isActive':'is_active',
        'draftYear':'draft_year',
        'mlbDebutDate':'debut_date',
        # 'mlbDebutTeam':'debut_team',
        'lastGame':'last_game_date',
        'recentTeam':'recent_team',
        'primaryNumber':'jersey',
        'primaryPosition':'position',
        'allNumbers':'all_jerseys',
        'allPositions':'all_positions',
        'allTeams':'all_teams'
        },inplace=True)
    if replace_existing is False:
        return df
    else:
        df.to_csv(BIOS_CSV,index=False)
